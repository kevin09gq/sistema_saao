## Esta consulta es para los empleados que no NSS:
SELECT 
e.id_empleado,
e.clave_empleado,
e.biometrico,
e.ap_paterno,
e.ap_materno, 
e.nombre,
d.nombre_departamento,
tb.descripcion as turno_base,
tb.hora_inicio as hora_inicio_base,
tb.hora_fin as hora_fin_base,
ts.descripcion as turno_sabado,
ts.hora_inicio as hora_inicio_sabado,
ts.hora_fin as hora_fin_sabado
FROM info_empleados e
INNER JOIN departamentos d on e.id_departamento = d.id_departamento
INNER JOIN empleado_turno et on et.id_empleado = e.id_empleado
INNER JOIN turnos tb on et.id_turno_base = tb.id_turno
INNER JOIN turnos ts on et.id_turno_sabado = ts.id_turno
WHERE e.id_status = 1 AND e.status_nss = 0
ORDER BY e.ap_paterno ASC
